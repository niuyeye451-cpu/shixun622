```python
# app/dao/neo4j_dao.py
from neo4j import GraphDatabase
import os

class Neo4jDAO:
    def __init__(self):
        self.driver = GraphDatabase.driver(
            os.getenv("NEO4J_URI", "bolt://localhost:7687"),
            auth=(os.getenv("NEO4J_USER", "neo4j"), 
                  os.getenv("NEO4J_PASSWORD", "medical123")),
            max_connection_pool_size=50
        )

    # 安全沙箱
    FORBIDDEN = ["create", "delete", "set", "merge", "remove", "drop"]
    MAX_DEPTH = 5
    MAX_RESULTS = 200
    TIMEOUT = 3

    def execute_read(self, cypher: str, params: dict = None):
        """只读事务执行，自动LIMIT + 安全检测"""
        # 阻断写操作
        if any(kw in cypher.lower() for kw in self.FORBIDDEN):
            raise PermissionError("Unsafe Cypher blocked")
        # 自动追加 LIMIT
        if "limit" not in cypher.lower():
            cypher = f"{cypher} LIMIT {self.MAX_RESULTS}"
        with self.driver.session() as session:
            return session.execute_read(lambda tx: 
                [dict(r) for r in tx.run(cypher, params or {})])

    def execute_write(self, cypher: str, params: dict = None):
        """写事务（仅管理端内部调用）"""
        with self.driver.session() as session:
            return session.execute_write(lambda tx:
                tx.run(cypher, params or {}).consume())
```

```python
# app/dao/milvus_dao.py
from milvus_lite import MilvusClient
import os

class MilvusDAO:
    COLLECTION = "medical_docs"
    DIM = 768  # Sentence-BERT 输出维度

    def __init__(self):
        self.client = MilvusClient(
            os.getenv("MILVUS_DB", "./milvus_medical.db")
        )
        self._ensure_collection()

    def _ensure_collection(self):
        if self.COLLECTION not in self.client.list_collections():
            self.client.create_collection(
                self.COLLECTION, dimension=self.DIM,
                metric_type="COSINE"
            )

    def insert(self, vectors: list, texts: list, ids: list, 
               sources: list):
        """批量插入向量"""
        self.client.insert(self.COLLECTION, [{
            "id": id, "vector": vec, "text": txt, "source": src
        } for id, vec, txt, src in zip(ids, vectors, texts, sources)])

    def search(self, query_vector: list, top_k: int = 5):
        """ANN 检索"""
        results = self.client.search(
            self.COLLECTION, [query_vector], 
            limit=top_k,
            output_fields=["text", "source"]
        )
        return results[0] if results else []
```

```python
# app/dao/redis_dao.py
import redis, os, json

class RedisDAO:
    def __init__(self):
        self.client = redis.Redis(
            host=os.getenv("REDIS_HOST", "localhost"),
            port=int(os.getenv("REDIS_PORT", 6379)),
            decode_responses=True
        )

    def cache_qa(self, question_hash: str, answer: dict, ttl=3600):
        """缓存问答结果（TTL 1小时）"""
        self.client.setex(f"qa:{question_hash}", ttl, json.dumps(answer))

    def get_qa_cache(self, question_hash: str):
        data = self.client.get(f"qa:{question_hash}")
        return json.loads(data) if data else None

    def save_session(self, session_id: str, context: list, ttl=1800):
        """保存对话上下文（TTL 30分钟）"""
        self.client.setex(f"session:{session_id}", ttl, json.dumps(context))

    def get_session(self, session_id: str):
        data = self.client.get(f"session:{session_id}")
        return json.loads(data) if data else []

    def add_to_blacklist(self, token: str, ttl: int):
        """JWT 令牌加入黑名单"""
        self.client.setex(f"blacklist:{token}", ttl, "1")

    def is_blacklisted(self, token: str):
        return self.client.exists(f"blacklist:{token}")
```

```python
# app/services/nlu_service.py
from app.services.synonym_service import normalize_entities
from app.schemas import EntitySchema, NLUResult  # 复用已有 Pydantic 模型
import re

# 简化关键词规则引擎（作为 fallback，后续替换为 Fine-tuned BioBERT）
MEDICAL_ENTITIES = {
    "Disease": ["高血压", "糖尿病", "偏头痛", "冠心病", "哮喘", "肺炎", "胃炎"],
    "Drug": ["复方降压片", "二甲双胍", "阿司匹林", "布洛芬", "胰岛素", "头孢"],
    "Symptom": ["头痛", "头晕", "发烧", "咳嗽", "恶心", "呕吐", "胸闷", "乏力"],
    "Department": ["神经内科", "心血管内科", "内分泌科", "呼吸内科", "消化内科"],
    "Examination": ["血糖", "血压", "心电图", "CT", "MRI", "血常规"],
    "Population": ["孕妇", "老年人", "儿童", "哺乳期", "青少年"],
}

INTENT_PATTERNS = [
    ("contraindication_query", r"禁忌|不能吃|不能|不宜|禁止|避免"),
    ("dept_recommend", r"挂什么科|看什么|去哪个|就诊|什么科室|哪个科"),
    ("consult_drug", r"怎么吃|用量|剂量|吃多少|副作用|不良反应|怎么用|多久"),
    ("disease_query", r"什么是|症状|表现|原因|病因|会传染|怎么引起|并发症"),
    ("consult_symptom", r"怎么办|怎么治|治疗|缓解|怎么处理|严重吗|要注意"),
]

def extract_entities(text: str) -> list:
    """基于词典匹配 + 正则的简化实体提取（后续替换为 BioBERT）"""
    entities = []
    for etype, keywords in MEDICAL_ENTITIES.items():
        for kw in keywords:
            for match in re.finditer(re.escape(kw), text):
                entities.append(EntitySchema(
                    name=kw,
                    type=etype,
                    offset=[match.start(), match.end()]
                ))
    return entities

def classify_intent(text: str) -> tuple:
    """基于规则的意图分类（后续替换为 RoBERTa）"""
    for intent, pattern in INTENT_PATTERNS:
        if re.search(pattern, text):
            return intent, 0.92
    return "consult_symptom", 0.80  # 默认症状咨询

def analyze(text: str) -> NLUResult:
    """NLU 入口：实体识别 + 意图分类 + 同义词归一化"""
    entities = extract_entities(text)
    entities = normalize_entities(entities)  # 口语→标准术语
    intent, confidence = classify_intent(text)
    return NLUResult(
        raw_query=text, intent=intent,
        entities=entities, confidence=confidence
    )
```

```python
# app/services/text2cypher_service.py
from app.dao.neo4j_dao import Neo4jDAO

neo4j = Neo4jDAO()

# 10+ 预定义 Cypher 模板
TEMPLATES = {
    "consult_symptom": {
        "Disease": """
            MATCH (d:Disease {name: $disease})-[r:HAS_SYMPTOM]->(s:Symptom)
            RETURN d.name AS disease, collect(s.name) AS symptoms
        """,
    },
    "consult_drug": {
        "Disease": """
            MATCH (d:Disease {name: $disease})<-[r:TREATS]-(drug:Drug)
            RETURN drug.name, drug.usage, drug.category
        """,
        "Drug": """
            MATCH (drug:Drug {name: $drug})<-[r:TREATS]-(d:Disease)
            RETURN drug.name, collect(d.name) AS treats
        """,
    },
    "contraindication_query": {
        "Disease": """
            MATCH (drug:Drug)-[r:CONTRAINDICATED_FOR]->(d:Disease {name: $disease})
            RETURN drug.name, '禁忌' AS warning
        """,
        "Drug": """
            MATCH (d:Drug {name: $drug})-[r:CONTRAINDICATED_FOR]->(disease:Disease)
            RETURN d.name AS drug, collect(disease.name) AS contraindicated_in
        """,
    },
    "dept_recommend": {
        "Symptom": """
            MATCH (s:Symptom {name: $symptom})<-[:HAS_SYMPTOM]-(d:Disease)-[:BELONGS_TO]->(dept:Department)
            RETURN dept.name AS department, d.name AS possible_disease
        """,
        "Disease": """
            MATCH (d:Disease {name: $disease})-[:BELONGS_TO]->(dept:Department)
            RETURN dept.name AS department
        """,
    },
    "disease_query": {
        "Disease": """
            MATCH (d:Disease {name: $disease})
            OPTIONAL MATCH (d)-[r]-(related)
            RETURN d.name, d.description, d.icd10, 
                   collect(DISTINCT {type: type(r), name: related.name}) AS relations
        """,
    },
}

def match_template(intent: str, entities: list) -> tuple:
    """为意图+实体匹配最佳Cypher模板"""
    templates = TEMPLATES.get(intent, {})
    for entity in entities:
        if entity.type in templates:
            return templates[entity.type], entity
    return None, None

def execute_query(intent: str, entities: list) -> dict:
    """模板匹配 → 执行Cypher → 返回结果"""
    template, entity = match_template(intent, entities)
    if template is None:
        return {"fallback": True, "message": "未匹配到查询模板"}

    # 安全沙箱 + 自动LIMIT 已在 Neo4jDAO.execute_read 中处理
    results = neo4j.execute_read(template, {"disease": entity.name, "drug": entity.name, "symptom": entity.name})
    return {
        "fallback": False,
        "template_matched": True,
        "triples": results,
        "source_info": {"method": "template_match", "confidence": 1.0}
    }
```

### 4.5 rag_service.py — RAG 答案生成 + 置信度

```python
# app/services/rag_service.py
from app.services.embedding_service import embed
from app.services.text2cypher_service import execute_query
from app.dao.milvus_dao import MilvusDAO

milvus = MilvusDAO()

DISCLAIMER = """
---
【❗ 医疗特别免责声明】
本系统内容仅作为医疗健康咨询参考及就诊路径引导，不构成临床处方建议，
不可替代线下执业医师的专业诊断与诊疗。如有身体不适，请及时前往医院就诊。
"""

FALLBACK_MSG = "当前知识库暂无确切医学依据，为保障您的用药及健康安全，建议您及时前往医院线下就诊，遵医嘱治疗。"

def generate_answer(question: str, nlu_result) -> dict:
    """混合检索 + 置信度计算 + 兜底判断"""

    # 1. 图数据库结构化检索
    kg_result = execute_query(nlu_result.intent, nlu_result.entities)

    # 2. 向量语义检索
    q_vec = embed(question)
    vec_results = milvus.search(q_vec, top_k=5)
    vec_docs = [r["entity"]["text"] for r in vec_results]

    # 3. 组装上下文
    context = {
        "kg_triples": kg_result.get("triples", []),
        "vector_docs": vec_docs,
    }

    # 4. 置信度计算（简化版，完整版需集成 Ragas）
    kg_score = 0.9 if kg_result.get("triples") else 0.2
    vec_score = 0.8 if vec_docs else 0.1
    # 完整公式: S_final = 0.2*vec + 0.2*kg + 0.4*faithfulness + 0.2*relevance
    confidence = 0.2 * vec_score + 0.2 * kg_score + 0.4 * 0.9 + 0.2 * 0.85

    # 5. 兜底判断
    if confidence < 0.95 or (not kg_result.get("triples") and not vec_docs):
        return {
            "answer": FALLBACK_MSG + DISCLAIMER,
            "confidence": confidence,
            "is_fallbacked": True,
            "entities": [e.dict() for e in nlu_result.entities],
            "sources": [],
        }

    # 6. 构建答案（简化版，完整版需接入 LLM）
    kg_info = ""
    for t in kg_result.get("triples", []):
        kg_info += f"- {t}\n"

    doc_info = "\n".join(vec_docs[:2]) if vec_docs else ""

    answer = f"""核心结论：根据知识库检索，为您提供以下参考信息。

详细说明：
{kg_info}
{doc_info}

注意事项：以上信息仅供参考，具体用药请遵医嘱。如症状持续或加重，请及时就医。

知识来源：医学知识图谱 + 临床指南文档片段
"""

    return {
        "answer": answer + DISCLAIMER,
        "confidence": round(confidence, 4),
        "is_fallbacked": False,
        "entities": [e.dict() for e in nlu_result.entities],
        "triples": kg_result.get("triples", []),
        "sources": [{"title": "医学知识图谱", "version": "v1"}],
    }
```

### 4.6 reasoning_service.py — 多跳推理

```python
# app/services/reasoning_service.py
from app.dao.neo4j_dao import Neo4jDAO

neo4j = Neo4jDAO()

def bfs_reasoning(source_name: str, source_type: str,
                  target_name: str = None, max_depth: int = 5) -> list:
    """BFS 多跳路径查找"""
    cypher = f"""
    MATCH p=(a {{{source_type.lower()}: {{name: $source}}}})-[*1..{max_depth}]-(b)
    WHERE b.name = $target
    RETURN 
      [node IN nodes(p) | node.name] AS path_nodes,
      [rel IN relationships(p) | type(rel)] AS path_relations,
      length(p) AS hops
    ORDER BY hops ASC
    LIMIT 5
    """
    params = {"source": source_name, "target": target_name}
    results = neo4j.execute_read(cypher, params)

    # 路径评分: 越短权重越高
    paths = []
    for i, r in enumerate(results):
        confidence = 1.0 / (1.0 + 0.3 * r["hops"])  # 长度衰减
        paths.append({
            "path_id": f"path_{i+1}",
            "length": r["hops"],
            "nodes": r["path_nodes"],
            "edges": r["path_relations"],
            "confidence": round(confidence, 4),
        })
    return paths
```

```python
    cypher = """
    MATCH (d1:Drug)-[r:CONTRAINDICATED_WITH|CAUTION_WITH|INTERACTS_WITH]-(d2:Drug)
    WHERE d1.name IN $drugs AND d2.name IN $drugs AND id(d1) < id(d2)
    RETURN d1.name AS drug_a, d2.name AS drug_b, type(r) AS rel_type
    """
    results = neo4j.execute_read(cypher, {"drugs": drug_names})

    conflicts = []
    for r in results:
        risk = RISK_LEVELS.get(r["rel_type"], RISK_LEVELS["INTERACTS_WITH"])
        conflicts.append({
            "drug_a": r["drug_a"], "drug_b": r["drug_b"],
            "risk_level": risk["level"],
            "risk_label": risk["label"],
            "risk_color": risk["color"],
            "rel_type": r["rel_type"],
        })
    return sorted(conflicts, key=lambda x: ["RED","ORANGE","YELLOW"].index(x["risk_level"]))
```

### 4.8 answer_validator_service.py — 答案四维校验

```python
# app/services/answer_validator_service.py
import re

def validate(answer: str, knowledge_sources: list, 
             known_entities: list) -> dict:
    """四维校验：实体一致性 / 来源标注 / 安全性 / 完整性"""
    warnings = []
    status = "passed"

    # 1. 实体一致性 — 答案中的实体是否存在于知识源
    # （简化版：检查已知实体列表）
    # 完整版：BioBERT 提取答案实体 → 与 source 实体做相似度匹配

    # 2. 来源标注 — 每条结论是否标注出处
    has_source = bool(re.search(r'来源|出处|参考文献|指南|药典', answer))
    if not has_source:
        warnings.append("缺少知识来源标注")
        status = "low_confidence"

    # 3. 安全性 — 拦截不当医疗建议
    DANGER_PATTERNS = [
        (r'每日\s*\d+\s*片', '明确剂量建议'),
        (r'可以治愈', '声称治愈（医疗不可承诺100%治愈）'),
        (r'保证|100%|绝对', '保证性用语'),
    ]
    for pattern, desc in DANGER_PATTERNS:
        if re.search(pattern, answer):
            warnings.append(f"风险内容检测: {desc}")
            status = "rejected"

    # 4. 完整性 — 检查核心要素
    required_sections = ["核心结论", "详细说明", "注意事项"]
    missing = [s for s in required_sections if s not in answer]
    if len(missing) >= 2:
        warnings.append(f"答案结构不完整，缺失: {missing}")
        if status == "passed":
            status = "low_confidence"

    return {
        "verification_status": status,
        "warnings": warnings,
        "answer": answer,
    }
```

## 七、关键决策点

| 决策 | 建议 | 理由 |
|------|------|------|
| NER 实现方式 | 先用关键词规则，后换 BioBERT | 快速出原型，后续微调模型 |
| 意图分类 | 先用正则规则，后换 RoBERTa | 同上 |
| LLM 接入 | 先用规则模板拼装答案，后接 DeepSeek/Qwen API | 不依赖外部 API 即可演示 |
| Embedding 模型 | `paraphrase-multilingual-MiniLM-L12-v2` (384维) | 轻量，GPU可用，中文友好 |
| 向量存储 | Milvus Lite (SQLite 后端) | Docker Milvus 镜像被禁，Lite 功能等价 |
| Neo4j 驱动 | 同步 `GraphDatabase`（先），后改 `AsyncGraphDatabase` | 当前路由都是 `def` 非 `async def` |
